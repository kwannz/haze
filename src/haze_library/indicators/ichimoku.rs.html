2#![allow(dead_code)]
3#![allow(clippy::needless_range_loop)]
4//
5// 一目均衡表是日本传统技术分析系统，包含 5 条线：
6// 1. Tenkan-sen（转换线）：9 周期高低平均
7// 2. Kijun-sen（基准线）：26 周期高低平均
8// 3. Senkou Span A（先行带 A）：转换线与基准线的平均，前移 26 周期
9// 4. Senkou Span B（先行带 B）：52 周期高低平均，前移 26 周期
10// 5. Chikou Span（延迟线）：收盘价，后移 26 周期
11//
12// "云"（Kumo）由先行带 A 和 B 形成，提供动态支撑/阻力区域
13

19/// Ichimoku Cloud 结构体
20#[derive(Debug, Clone)]
21pub struct IchimokuCloud {
22    pub tenkan_sen: Vec<f64>,    // 转换线
23    pub kijun_sen: Vec<f64>,     // 基准线
24    pub senkou_span_a: Vec<f64>, // 先行带 A
25    pub senkou_span_b: Vec<f64>, // 先行带 B
26    pub chikou_span: Vec<f64>,   // 延迟线
27}

29#[inline]
30fn validate_ichimoku_lengths(ichimoku: &IchimokuCloud) -> HazeResult<usize> {
31    let n = ichimoku.tenkan_sen.len();
32    if n == 0 {
33        return Err(HazeError::EmptyInput {
34            name: "ichimoku.tenkan_sen",
35        });
36    }
37    if ichimoku.kijun_sen.len() != n {
38        return Err(HazeError::LengthMismatch {
39            name1: "ichimoku.tenkan_sen",
40            len1: n,
41            name2: "ichimoku.kijun_sen",
42            len2: ichimoku.kijun_sen.len(),
43        });
44    }
45    if ichimoku.senkou_span_a.len() != n {
46        return Err(HazeError::LengthMismatch {
47            name1: "ichimoku.tenkan_sen",
48            len1: n,
49            name2: "ichimoku.senkou_span_a",
50            len2: ichimoku.senkou_span_a.len(),
51        });
52    }
53    if ichimoku.senkou_span_b.len() != n {
54        return Err(HazeError::LengthMismatch {
55            name1: "ichimoku.tenkan_sen",
56            len1: n,
57            name2: "ichimoku.senkou_span_b",
58            len2: ichimoku.senkou_span_b.len(),
59        });
60    }
61    if ichimoku.chikou_span.len() != n {
62        return Err(HazeError::LengthMismatch {
63            name1: "ichimoku.tenkan_sen",
64            len1: n,
65            name2: "ichimoku.chikou_span",
66            len2: ichimoku.chikou_span.len(),
67        });
68    }
69    Ok(n)
70}

72/// 计算 Ichimoku Cloud（一目均衡表）
73///
74/// - `high`: 高价序列
75/// - `low`: 低价序列
76/// - `close`: 收盘价序列
77/// - `tenkan_period`: 转换线周期（默认 9）
78/// - `kijun_period`: 基准线周期（默认 26）
79/// - `senkou_b_period`: 先行带 B 周期（默认 52）
80///
81/// 返回：Ichimoku Cloud 结构体
82///
83/// # 算法
84/// 1. Tenkan-sen = (9-period high + 9-period low) / 2
85/// 2. Kijun-sen = (26-period high + 26-period low) / 2
86/// 3. Senkou Span A = (Tenkan + Kijun) / 2, shifted forward 26
87/// 4. Senkou Span B = (52-period high + 52-period low) / 2, shifted forward 26
88/// 5. Chikou Span = Close, shifted backward 26
89///
90/// # 示例
91/// ```rust,no_run
92/// use haze_library::indicators::ichimoku::ichimoku_cloud;
93///
94/// let high = vec![110.0; 100];
95/// let low = vec![100.0; 100];
96/// let close = vec![105.0; 100];
97///
98/// let ichimoku = ichimoku_cloud(&high, &low, &close, 9, 26, 52).unwrap();
99/// assert_eq!(ichimoku.tenkan_sen.len(), close.len());
100/// ```
101pub fn ichimoku_cloud(
102    high: &[f64],
103    low: &[f64],
104    close: &[f64],
105    tenkan_period: usize,
106    kijun_period: usize,
107    senkou_b_period: usize,
108) -> HazeResult<IchimokuCloud> {
109    validate_not_empty(high, "high")?;
110    validate_lengths_match(&[(high, "high"), (low, "low"), (close, "close")])?;
111    let n = high.len();
112    validate_period(tenkan_period, n)?;
113    validate_period(kijun_period, n)?;
114    validate_period(senkou_b_period, n)?;
115
116    // 1. Tenkan-sen (Conversion Line): (9-period high + low) / 2
117    let tenkan_sen = calc_donchian_midline(high, low, tenkan_period);
118
119    // 2. Kijun-sen (Base Line): (26-period high + low) / 2
120    let kijun_sen = calc_donchian_midline(high, low, kijun_period);
121
122    // 3. Senkou Span A (Leading Span A): (Tenkan + Kijun) / 2, shifted +26
123    let mut senkou_span_a = init_result!(n);
124    for i in 0..n {
125        if !tenkan_sen[i].is_nan() && !kijun_sen[i].is_nan() {
126            let value = (tenkan_sen[i] + kijun_sen[i]) / 2.0;
127            // 前移 26 周期（如果在范围内）
128            let future_idx = i + kijun_period;
129            if future_idx < n {
130                senkou_span_a[future_idx] = value;
131            }
132        }
133    }
134
135    // 4. Senkou Span B (Leading Span B): (52-period high + low) / 2, shifted +26
136    let span_b_midline = calc_donchian_midline(high, low, senkou_b_period);
137    let mut senkou_span_b = init_result!(n);
138    for i in 0..n {
139        if !span_b_midline[i].is_nan() {
140            let future_idx = i + kijun_period;
141            if future_idx < n {
142                senkou_span_b[future_idx] = span_b_midline[i];
143            }
144        }
145    }
146
147    // 5. Chikou Span (Lagging Span): Close price, shifted -26
148    let mut chikou_span = init_result!(n);
149    chikou_span[..(n - kijun_period)].copy_from_slice(&close[kijun_period..n]);
150
151    Ok(IchimokuCloud {
152        tenkan_sen,
153        kijun_sen,
154        senkou_span_a,
155        senkou_span_b,
156        chikou_span,
157    })
158}

160/// 计算 Donchian 中线（高低价的平均）
161///
162/// 这是 Ichimoku 各线的核心计算逻辑
163fn calc_donchian_midline(high: &[f64], low: &[f64], period: usize) -> Vec<f64> {
164    let n = high.len();
165    let mut result = init_result!(n);
166
167    let high_max = rolling_max(high, period);
168    let low_min = rolling_min(low, period);
169
170    for i in 0..n {
171        if !high_max[i].is_nan() && !low_min[i].is_nan() {
172            result[i] = (high_max[i] + low_min[i]) / 2.0;
173        }
174    }
175
176    result
177}

179/// Ichimoku 信号类型
180#[derive(Debug, Clone, Copy, PartialEq)]
181pub enum IchimokuSignal {
182    StrongBullish, // 强看涨：价格在云上方，云为绿色（Span A > Span B），Chikou 在价格上方
183    Bullish,       // 看涨：价格在云上方
184    Neutral,       // 中性：价格在云内
185    Bearish,       // 看跌：价格在云下方
186    StrongBearish, // 强看跌：价格在云下方，云为红色（Span A < Span B），Chikou 在价格下方
187}

189/// 生成 Ichimoku 信号
190///
191/// - `close`: 收盘价序列
192/// - `ichimoku`: Ichimoku Cloud 结构体
193///
194/// 返回：每个价格对应的信号向量
195pub fn ichimoku_signals(
196    close: &[f64],
197    ichimoku: &IchimokuCloud,
198) -> HazeResult<Vec<IchimokuSignal>> {
199    validate_not_empty(close, "close")?;
200    let n = validate_ichimoku_lengths(ichimoku)?;
201    if close.len() != n {
202        return Err(HazeError::LengthMismatch {
203            name1: "close",
204            len1: close.len(),
205            name2: "ichimoku.tenkan_sen",
206            len2: n,
207        });
208    }
209    let mut signals = Vec::with_capacity(n);
210
211    for i in 0..n {
212        let price = close[i];
213        let span_a = ichimoku.senkou_span_a[i];
214        let span_b = ichimoku.senkou_span_b[i];
215        let chikou = ichimoku.chikou_span[i];
216
217        if span_a.is_nan() || span_b.is_nan() {
218            signals.push(IchimokuSignal::Neutral);
219            continue;
220        }
221
222        let cloud_top = span_a.max(span_b);
223        let cloud_bottom = span_a.min(span_b);
224        let is_green_cloud = span_a > span_b;
225
226        // 判断价格位置
227        let signal = if price > cloud_top {
228            // 价格在云上方
229            if is_green_cloud && !chikou.is_nan() && chikou > price {
230                IchimokuSignal::StrongBullish
231            } else {
232                IchimokuSignal::Bullish
233            }
234        } else if price < cloud_bottom {
235            // 价格在云下方
236            if !is_green_cloud && !chikou.is_nan() && chikou < price {
237                IchimokuSignal::StrongBearish
238            } else {
239                IchimokuSignal::Bearish
240            }
241        } else {
242            // 价格在云内
243            IchimokuSignal::Neutral
244        };
245
246        signals.push(signal);
247    }
248
249    Ok(signals)
250}

252/// TK Cross（转换线与基准线交叉）信号
253///
254/// - `ichimoku`: Ichimoku Cloud 结构体
255///
256/// 返回：交叉信号向量（1.0=金叉，-1.0=死叉，0.0=无交叉）
257///
258/// # 算法
259/// - 金叉（Bullish TK Cross）：Tenkan 向上穿过 Kijun
260/// - 死叉（Bearish TK Cross）：Tenkan 向下穿过 Kijun
261pub fn ichimoku_tk_cross(ichimoku: &IchimokuCloud) -> HazeResult<Vec<f64>> {
262    let n = validate_ichimoku_lengths(ichimoku)?;
263    let mut signals = vec![0.0; n];
264
265    for i in 1..n {
266        let prev_tenkan = ichimoku.tenkan_sen[i - 1];
267        let prev_kijun = ichimoku.kijun_sen[i - 1];
268        let curr_tenkan = ichimoku.tenkan_sen[i];
269        let curr_kijun = ichimoku.kijun_sen[i];
270
271        if prev_tenkan.is_nan()
272            || prev_kijun.is_nan()
273            || curr_tenkan.is_nan()
274            || curr_kijun.is_nan()
275        {
276            continue;
277        }
278
279        // 金叉：前一根 Tenkan <= Kijun，当前 Tenkan > Kijun
280        if prev_tenkan <= prev_kijun && curr_tenkan > curr_kijun {
281            signals[i] = 1.0;
282        }
283        // 死叉：前一根 Tenkan >= Kijun，当前 Tenkan < Kijun
284        else if prev_tenkan >= prev_kijun && curr_tenkan < curr_kijun {
285            signals[i] = -1.0;
286        }
287    }
288
289    Ok(signals)
290}

292/// 云厚度（Cloud Thickness）
293///
294/// 云厚度越大，支撑/阻力越强
295///
296/// - `ichimoku`: Ichimoku Cloud 结构体
297///
298/// 返回：云厚度向量（绝对值）
299pub fn cloud_thickness(ichimoku: &IchimokuCloud) -> HazeResult<Vec<f64>> {
300    let n = validate_ichimoku_lengths(ichimoku)?;
301    let mut thickness = Vec::with_capacity(n);
302
303    for i in 0..n {
304        let span_a = ichimoku.senkou_span_a[i];
305        let span_b = ichimoku.senkou_span_b[i];
306
307        if span_a.is_nan() || span_b.is_nan() {
308            thickness.push(f64::NAN);
309        } else {
310            thickness.push((span_a - span_b).abs());
311        }
312    }
313
314    Ok(thickness)
315}

317/// Ichimoku 云颜色（Cloud Color）
318///
319/// - `ichimoku`: Ichimoku Cloud 结构体
320///
321/// 返回：云颜色向量（1.0=绿色/看涨，-1.0=红色/看跌，0.0=中性）
322pub fn cloud_color(ichimoku: &IchimokuCloud) -> HazeResult<Vec<f64>> {
323    let n = validate_ichimoku_lengths(ichimoku)?;
324    let mut colors = Vec::with_capacity(n);
325
326    for i in 0..n {
327        let span_a = ichimoku.senkou_span_a[i];
328        let span_b = ichimoku.senkou_span_b[i];
329
330        if span_a.is_nan() || span_b.is_nan() {
331            colors.push(0.0);
332        } else if span_a > span_b {
333            colors.push(1.0); // 绿云（看涨）
334        } else if span_a < span_b {
335            colors.push(-1.0); // 红云（看跌）
336        } else {
337            colors.push(0.0); // 中性
338        }
339    }
340
341    Ok(colors)
342}

349    fn test_ichimoku_cloud_basic() {
350        let high = vec![110.0; 100];
351        let low = vec![100.0; 100];
352        let close = vec![105.0; 100];
353
354        let ichimoku = ichimoku_cloud(&high, &low, &close, 9, 26, 52).unwrap();
355
356        // 在横盘市场中，Tenkan 和 Kijun 应为 (110+100)/2 = 105
357        let valid_idx = 26; // 第一个有效的索引（Kijun 周期）
358        assert!((ichimoku.tenkan_sen[valid_idx] - 105.0).abs() < 0.1);
359        assert!((ichimoku.kijun_sen[valid_idx] - 105.0).abs() < 0.1);
360    }

363    fn test_tk_cross() {
364        // Need at least 52 data points for senkou_span_b_period=52
365        let high: Vec<f64> = (0..60).map(|i| 100.0 + i as f64).collect();
366        let low: Vec<f64> = high.iter().map(|&h| h - 2.0).collect();
367        let close: Vec<f64> = high.iter().map(|&h| h - 1.0).collect();
368
369        let ichimoku = ichimoku_cloud(&low, &high, &close, 9, 26, 52).unwrap();
370        let crosses = ichimoku_tk_cross(&ichimoku).unwrap();
371
372        // 在上升趋势中，应该有金叉信号
373        let has_golden_cross = crosses.iter().any(|&c| c > 0.0);
374        assert!(has_golden_cross || crosses.iter().all(|&c| c == 0.0)); // 可能周期太短没交叉
375    }
