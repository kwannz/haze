1//! Volatility Indicators Module
2//!
3//! # Overview
4//! This module provides volatility-based technical indicators that measure the
5//! degree of price variation over time. Volatility indicators are essential for
6//! risk management, position sizing, and identifying potential breakout conditions.
7//!
8//! # Error Handling
9//! All public functions return `HazeResult<T>` with proper error types:
10//! - `HazeError::EmptyInput` - when input arrays are empty
11//! - `HazeError::LengthMismatch` - when input arrays have different lengths
12//! - `HazeError::InvalidPeriod` - when period is 0 or exceeds data length
13//! - `HazeError::InsufficientData` - when data length is insufficient for the operation
14//! - `HazeError::ParameterOutOfRange` - when parameters like multipliers are invalid
15//!
16//! # Available Functions
17//! - [`true_range`] - True Range (maximum of H-L, |H-Prev_C|, |L-Prev_C|)
18//! - [`atr`] - Average True Range (smoothed TR using Wilder's RMA)
19//! - [`natr`] - Normalized ATR (ATR as percentage of close price)
20//! - [`bollinger_bands`] - Bollinger Bands (SMA +/- standard deviation bands)
21//! - [`keltner_channel`] - Keltner Channel (EMA +/- ATR-based bands)
22//! - [`donchian_channel`] - Donchian Channel (highest high / lowest low bands)
23//! - [`chandelier_exit`] - Chandelier Exit (ATR-based trailing stop levels)
24//! - [`historical_volatility`] - Historical Volatility (annualized log returns std)
25//! - [`ulcer_index`] - Ulcer Index (drawdown-based risk measure)
26//! - [`mass_index`] - Mass Index (range expansion indicator)
27//!
28//! # Examples
29//! ```rust
30//! use haze_library::indicators::volatility::{atr, bollinger_bands};
31//!
32//! let high = vec![102.0, 105.0, 104.0, 106.0, 108.0];
33//! let low = vec![99.0, 101.0, 100.0, 102.0, 104.0];
34//! let close = vec![101.0, 103.0, 102.0, 105.0, 107.0];
35//!
36//! // Calculate 3-period ATR (returns Result)
37//! let atr_values = atr(&high, &low, &close, 3).unwrap();
38//!
39//! // Calculate Bollinger Bands with 3-period SMA and 2 std dev
40//! let (upper, middle, lower) = bollinger_bands(&close, 3, 2.0).unwrap();
41//! ```
42//!
43//! # Performance Characteristics
44//! - ATR: O(n) using incremental Wilder's smoothing (RMA)
45//! - Bollinger Bands: O(n) with rolling statistics computation
46//! - Donchian Channel: O(n) using monotonic deque for rolling max/min
47//!
48//! # Cross-References
49//! - [`crate::indicators::trend`] - SuperTrend uses ATR for band calculation
50//! - [`crate::utils::stats`] - Standard deviation and rolling functions
51//! - [`crate::utils::ma`] - EMA/SMA for band center lines

67/// True Range (TR)
68///
69/// Calculates the true range, which is the greatest of:
70/// - Current High - Current Low
71/// - |Current High - Previous Close|
72/// - |Current Low - Previous Close|
73///
74/// # Algorithm
75/// ```text
76/// TR = MAX(high - low, ABS(high - prev_close), ABS(low - prev_close))
77/// ```
78///
79/// # Parameters
80/// - `high`: High price series
81/// - `low`: Low price series
82/// - `close`: Close price series
83/// - `drift`: Lookback period for previous close (typically 1)
84///
85/// # Returns
86/// - `Ok(Vec<f64>)`: Vector of same length as input, first `drift` values are NaN
87///
88/// # Errors
89/// - [`HazeError::EmptyInput`]: Any input array is empty
90/// - [`HazeError::LengthMismatch`]: Input arrays have different lengths
91/// - [`HazeError::InvalidPeriod`]: drift is 0
92/// - [`HazeError::InsufficientData`]: drift >= data length
93///
94/// # Example
95/// ```rust
96/// use haze_library::indicators::volatility::true_range;
97///
98/// let high = vec![102.0, 105.0, 104.0];
99/// let low = vec![99.0, 101.0, 100.0];
100/// let close = vec![101.0, 103.0, 102.0];
101///
102/// let tr = true_range(&high, &low, &close, 1).unwrap();
103/// assert!(tr[0].is_nan()); // No previous close available
104/// assert_eq!(tr[1], 4.0);  // MAX(4, 4, 0) = 4.0
105/// ```
106pub fn true_range(high: &[f64], low: &[f64], close: &[f64], drift: usize) -> HazeResult<Vec<f64>> {
107    // Validate inputs
108    validate_not_empty(high, "high")?;
109    validate_lengths_match(&[(high, "high"), (low, "low"), (close, "close")])?;
110
111    let n = high.len();
112
113    // Validate drift parameter
114    if drift == 0 {
115        return Err(HazeError::InvalidPeriod {
116            period: drift,
117            data_len: n,
118        });
119    }
120    if drift >= n {
121        return Err(HazeError::InsufficientData {
122            required: drift + 1,
123            actual: n,
124        });
125    }
126
127    let mut result = init_result!(n);
128
129    // Calculate TR from index `drift` onwards (earlier values have no previous close)
130    for i in drift..n {
131        let prev_close = close[i - drift];
132        let tr1 = high[i] - low[i];
133        let tr2 = (high[i] - prev_close).abs();
134        let tr3 = (low[i] - prev_close).abs();
135        result[i] = tr1.max(tr2).max(tr3);
136    }
137
138    Ok(result)
139}

141/// Calculates the Average True Range (ATR).
142///
143/// ATR is a technical analysis volatility indicator that measures market volatility
144/// by decomposing the entire range of an asset price for a given period. It was
145/// developed by J. Welles Wilder Jr. and is widely used for position sizing,
146/// stop-loss placement, and volatility-based trading strategies.
147///
148/// # Formula
149/// ```text
150/// 1. Calculate True Range (TR) for each period:
151///    TR = MAX(high - low, |high - prev_close|, |low - prev_close|)
152///
153/// 2. Initial ATR (at period n):
154///    ATR[n] = SMA(TR[1..=n])
155///
156/// 3. Subsequent ATR values (Wilder's smoothing/RMA):
157///    ATR[i] = (ATR[i-1] * (period-1) + TR[i]) / period
158///
159/// Note: This uses Wilder's smoothing, which is similar to an exponential
160/// moving average but with α = 1/period instead of 2/(period+1).
161/// ```
162///
163/// # Arguments
164/// * `high` - High price series
165/// * `low` - Low price series
166/// * `close` - Close price series
167/// * `period` - Smoothing period for ATR calculation (typically 14)
168///
169/// # Returns
170/// Vector of ATR values. First `period` values will be NaN, as ATR requires
171/// a full period of TR values plus one for the initial average.
172///
173/// # Errors
174/// Returns error if:
175/// - [`HazeError::EmptyInput`] - Any input array is empty
176/// - [`HazeError::LengthMismatch`] - Input arrays have different lengths
177/// - [`HazeError::InvalidPeriod`] - `period` is 0
178/// - [`HazeError::InsufficientData`] - `period` >= length of data
179///
180/// # Examples
181/// ```rust
182/// use haze_library::indicators::volatility::atr;
183///
184/// let high = vec![102.0, 105.0, 104.0, 106.0, 108.0, 110.0, 112.0, 114.0,
185///                 116.0, 118.0, 120.0, 122.0, 124.0, 126.0, 128.0];
186/// let low = vec![99.0, 101.0, 100.0, 102.0, 104.0, 106.0, 108.0, 110.0,
187///                112.0, 114.0, 116.0, 118.0, 120.0, 122.0, 124.0];
188/// let close = vec![101.0, 103.0, 102.0, 105.0, 107.0, 109.0, 111.0, 113.0,
189///                  115.0, 117.0, 119.0, 121.0, 123.0, 125.0, 127.0];
190///
191/// // Calculate 14-period ATR
192/// let atr_values = atr(&high, &low, &close, 14).unwrap();
193///
194/// // First 14 values are NaN (warmup period)
195/// assert!(atr_values[13].is_nan());
196/// // First valid ATR at index 14
197/// assert!(!atr_values[14].is_nan());
198/// assert!(atr_values[14] > 0.0);
199/// ```
200///
201/// # Performance
202/// - Time complexity: O(n) where n = data.len()
203/// - Space complexity: O(n) for TR and result vectors
204/// - Single-pass algorithm using incremental Wilder's smoothing
205///
206/// # Trading Applications
207/// - **Volatility Measurement**: Higher ATR = higher volatility
208/// - **Position Sizing**: Risk a fixed multiple of ATR per trade
209/// - **Stop-Loss Placement**: Place stops at 2-3x ATR from entry
210/// - **Breakout Confirmation**: Rising ATR confirms breakout strength
211/// - **Trailing Stops**: Use in Chandelier Exit and similar systems
212///
213/// # Implementation Notes
214/// This implementation follows TA-Lib conventions:
215/// - TR`[0]` is ignored in ATR calculation
216/// - Initial ATR uses simple average of TR[1..=period]
217/// - Subsequent values use Wilder's smoothing (RMA)
218///
219/// # References
220/// - Wilder, J. W. (1978). New Concepts in Technical Trading Systems
221/// - Standard period: 14 (can be adjusted for different timeframes)
222///
223/// # See Also
224/// - [`true_range`] - Underlying TR calculation
225/// - [`natr`] - Normalized ATR (as percentage of price)
226/// - [`chandelier_exit`] - ATR-based trailing stop
227/// - [`keltner_channel`] - ATR-based volatility bands
228pub fn atr(high: &[f64], low: &[f64], close: &[f64], period: usize) -> HazeResult<Vec<f64>> {
229    // Validate inputs
230    validate_not_empty(high, "high")?;
231    validate_lengths_match(&[(high, "high"), (low, "low"), (close, "close")])?;
232
233    let n = high.len();
234
235    // Validate period parameter
236    if period == 0 {
237        return Err(HazeError::InvalidPeriod {
238            period,
239            data_len: n,
240        });
241    }
242    if period >= n {
243        return Err(HazeError::InsufficientData {
244            required: period + 1,
245            actual: n,
246        });
247    }
248
249    let tr = true_range(high, low, close, 1)?;
250    let mut result = init_result!(n);
251
252    // TA-Lib compatible: ignore TR[0], initial ATR = mean(TR[1..=period])
253    let sum = kahan_sum(&tr[1..=period]);
254    let period_f = period as f64;
255    result[period] = sum / period_f;
256
257    // Wilder's smoothing (RMA)
258    for i in (period + 1)..n {
259        result[i] = (result[i - 1] * (period_f - 1.0) + tr[i]) / period_f;
260    }
261
262    Ok(result)
263}

265/// NATR - Normalized Average True Range
266///
267/// Expresses ATR as a percentage of the close price, making it comparable
268/// across different price levels and instruments.
269///
270/// # Algorithm
271/// ```text
272/// NATR = (ATR / close) * 100
273/// ```
274///
275/// # Parameters
276/// - `high`: High price series
277/// - `low`: Low price series
278/// - `close`: Close price series
279/// - `period`: ATR period (typically 14)
280///
281/// # Returns
282/// - `Ok(Vec<f64>)`: NATR values as percentages
283///
284/// # Errors
285/// - [`HazeError::EmptyInput`]: Any input array is empty
286/// - [`HazeError::LengthMismatch`]: Input arrays have different lengths
287/// - [`HazeError::InvalidPeriod`]: period is 0 or >= data length
288///
289/// # Example
290/// ```rust
291/// use haze_library::indicators::volatility::natr;
292///
293/// let high = vec![102.0, 105.0, 104.0, 106.0, 108.0];
294/// let low = vec![99.0, 101.0, 100.0, 102.0, 104.0];
295/// let close = vec![101.0, 103.0, 102.0, 105.0, 107.0];
296///
297/// let natr_values = natr(&high, &low, &close, 3).unwrap();
298/// // NATR is expressed as percentage (e.g., 3.5 means 3.5%)
299/// ```
300pub fn natr(high: &[f64], low: &[f64], close: &[f64], period: usize) -> HazeResult<Vec<f64>> {
301    // Validate close is not empty (atr will validate the rest)
302    validate_not_empty(close, "close")?;
303
304    let atr_values = atr(high, low, close, period)?;
305
306    let result = atr_values
307        .iter()
308        .zip(close)
309        .map(|(&a, &c)| {
310            if a.is_nan() || is_zero(c) {
311                f64::NAN
312            } else {
313                (a / c) * 100.0
314            }
315        })
316        .collect();
317
318    Ok(result)
319}

321/// Calculates Bollinger Bands.
322///
323/// Bollinger Bands are a volatility indicator consisting of a middle band (SMA)
324/// with upper and lower bands at a specified number of standard deviations away.
325/// Developed by John Bollinger, they help identify overbought/oversold conditions,
326/// volatility expansion/contraction, and potential reversal points.
327///
328/// # Formula
329/// ```text
330/// 1. Middle Band (BASIS):
331///    MB = SMA(close, period)
332///
333/// 2. Standard Deviation:
334///    σ = StdDev(close, period)    [Population StdDev]
335///
336/// 3. Upper Band:
337///    UB = MB + (σ * std_multiplier)
338///
339/// 4. Lower Band:
340///    LB = MB - (σ * std_multiplier)
341///
342/// Typical Settings:
343/// - Period: 20
344/// - Multiplier: 2.0 (captures ~95% of price action)
345/// ```
346///
347/// # Arguments
348/// * `close` - Close price series
349/// * `period` - SMA period for middle band (commonly 20)
350/// * `std_multiplier` - Standard deviation multiplier for band width (commonly 2.0)
351///
352/// # Returns
353/// Tuple of three vectors: `(upper_band, middle_band, lower_band)`
354/// - `upper_band`: Upper Bollinger Band
355/// - `middle_band`: Middle band (SMA)
356/// - `lower_band`: Lower Bollinger Band
357///
358/// All vectors have the same length as input. First `period - 1` values are NaN.
359///
360/// # Errors
361/// Returns error if:
362/// - [`HazeError::EmptyInput`] - `close` array is empty
363/// - [`HazeError::InvalidPeriod`] - `period` is 0 or > data length
364/// - [`HazeError::ParameterOutOfRange`] - `std_multiplier` < 0
365///
366/// # Examples
367/// ```rust
368/// use haze_library::indicators::volatility::bollinger_bands;
369///
370/// let close = vec![
371///     100.0, 101.0, 102.0, 103.0, 104.0, 105.0, 104.5, 104.0,
372///     105.0, 106.0, 107.0, 108.0, 109.0, 110.0, 111.0, 112.0,
373///     113.0, 114.0, 115.0, 116.0, 117.0
374/// ];
375///
376/// // Standard Bollinger Bands (20-period, 2 std dev)
377/// let (upper, middle, lower) = bollinger_bands(&close, 20, 2.0).unwrap();
378///
379/// // Check band values
380/// assert!((middle[19] - 107.475).abs() < 1e-10);  // SMA of first 20 values
381/// assert!(upper[19] > middle[19]);
382/// assert!(lower[19] < middle[19]);
383///
384/// // Detect squeeze (bands narrowing)
385/// let bandwidth = (upper[19] - lower[19]) / middle[19];
386/// if bandwidth < 0.05 {  // Less than 5%
387///     println!("Bollinger Squeeze detected - volatility contraction");
388/// }
389/// ```
390///
391/// # Performance
392/// - Time complexity: O(n) where n = data.len()
393/// - Space complexity: O(n) for SMA and StdDev calculations
394/// - Efficient rolling statistics implementation
395///
396/// # Trading Strategies
397///
398/// **1. Bollinger Bounce**
399/// - Buy when price touches lower band
400/// - Sell when price touches upper band
401/// - Works best in ranging markets
402///
403/// **2. Bollinger Squeeze**
404/// - Narrow bands indicate low volatility
405/// - Often precedes significant price moves
406/// - Breakout direction determined by first band touch
407///
408/// **3. Band Walk**
409/// - Price consistently touching upper band = strong uptrend
410/// - Price consistently touching lower band = strong downtrend
411/// - Stay with trend until bands widen significantly
412///
413/// **4. %B Indicator**
414/// - %B = (close - lower) / (upper - lower)
415/// - %B > 1: Price above upper band
416/// - %B < 0: Price below lower band
417/// - %B = 0.5: Price at middle band
418///
419/// # Implementation Notes
420/// - Uses population standard deviation (divides by n, not n-1)
421/// - TA-Lib compatible calculation
422/// - Bands widen during volatility expansion
423/// - Bands narrow during volatility contraction
424///
425/// # References
426/// - Bollinger, J. (2001). Bollinger on Bollinger Bands
427/// - Standard parameters: 20-period SMA, 2.0 standard deviations
428/// - Covers approximately 95% of price action under normal distribution
429///
430/// # See Also
431/// - [`keltner_channel`] - Similar concept using ATR instead of StdDev
432/// - [`donchian_channel`] - Price channel using highest/lowest
433/// - [`sma`] - Simple Moving Average (middle band basis)
434/// - [`stdev_population`] - Population standard deviation calculation
435pub fn bollinger_bands(
436    close: &[f64],
437    period: usize,
438    std_multiplier: f64,
439) -> HazeResult<(Vec<f64>, Vec<f64>, Vec<f64>)> {
440    // Validate inputs
441    validate_not_empty(close, "close")?;
442    validate_period(period, close.len())?;
443
444    // Validate std_multiplier
445    if std_multiplier < 0.0 {
446        return Err(HazeError::ParameterOutOfRange {
447            name: "std_multiplier",
448            value: std_multiplier,
449            min: 0.0,
450            max: f64::INFINITY,
451        });
452    }
453
454    // 单次遍历同时计算 mean 和 stdev（Welford 算法优化）
455    let (middle, std) = mean_and_stdev_population(close, period);
456
457    let upper: Vec<f64> = middle
458        .iter()
459        .zip(&std)
460        .map(|(&m, &s)| {
461            if m.is_nan() || s.is_nan() {
462                f64::NAN
463            } else {
464                m + s * std_multiplier
465            }
466        })
467        .collect();
468
469    let lower: Vec<f64> = middle
470        .iter()
471        .zip(&std)
472        .map(|(&m, &s)| {
473            if m.is_nan() || s.is_nan() {
474                f64::NAN
475            } else {
476                m - s * std_multiplier
477            }
478        })
479        .collect();
480
481    Ok((upper, middle, lower))
482}

484/// Keltner Channel
485///
486/// A volatility-based envelope indicator using EMA as the center line
487/// and ATR for band width calculation.
488///
489/// # Algorithm
490/// ```text
491/// Middle Line = EMA(close, period)
492/// Upper Line  = Middle Line + (ATR * multiplier)
493/// Lower Line  = Middle Line - (ATR * multiplier)
494/// ```
495///
496/// # Parameters
497/// - `high`: High price series
498/// - `low`: Low price series
499/// - `close`: Close price series
500/// - `period`: EMA period for middle line (typically 20)
501/// - `atr_period`: ATR period for band width (typically 10)
502/// - `multiplier`: ATR multiplier for band distance (typically 2.0)
503///
504/// # Returns
505/// - `Ok((upper, middle, lower))`: Three vectors of channel values
506///
507/// # Errors
508/// - [`HazeError::EmptyInput`]: Any input array is empty
509/// - [`HazeError::LengthMismatch`]: Input arrays have different lengths
510/// - [`HazeError::InvalidPeriod`]: period or atr_period is 0
511/// - [`HazeError::InsufficientData`]: atr_period >= data length
512///
513/// # Example
514/// ```rust
515/// use haze_library::indicators::volatility::keltner_channel;
516///
517/// let high = vec![102.0, 105.0, 104.0, 106.0, 108.0, 110.0];
518/// let low = vec![99.0, 101.0, 100.0, 102.0, 104.0, 106.0];
519/// let close = vec![101.0, 103.0, 102.0, 105.0, 107.0, 109.0];
520///
521/// let (upper, middle, lower) = keltner_channel(&high, &low, &close, 3, 3, 2.0).unwrap();
522/// ```
523pub fn keltner_channel(
524    high: &[f64],
525    low: &[f64],
526    close: &[f64],
527    period: usize,
528    atr_period: usize,
529    multiplier: f64,
530) -> HazeResult<(Vec<f64>, Vec<f64>, Vec<f64>)> {
531    // Validate inputs
532    validate_not_empty(high, "high")?;
533    validate_lengths_match(&[(high, "high"), (low, "low"), (close, "close")])?;
534
535    let n = high.len();
536    validate_period(period, n)?;
537
538    // Validate ATR period
539    if atr_period == 0 {
540        return Err(HazeError::InvalidPeriod {
541            period: atr_period,
542            data_len: n,
543        });
544    }
545    if atr_period >= n {
546        return Err(HazeError::InsufficientData {
547            required: atr_period + 1,
548            actual: n,
549        });
550    }
551
552    // Validate multiplier
553    if multiplier < 0.0 {
554        return Err(HazeError::ParameterOutOfRange {
555            name: "multiplier",
556            value: multiplier,
557            min: 0.0,
558            max: f64::INFINITY,
559        });
560    }
561
562    let middle = ema(close, period)?;
563    let atr_values = atr(high, low, close, atr_period)?;
564
565    let upper: Vec<f64> = middle
566        .iter()
567        .zip(&atr_values)
568        .map(|(&m, &a)| {
569            if m.is_nan() || a.is_nan() {
570                f64::NAN
571            } else {
572                m + a * multiplier
573            }
574        })
575        .collect();
576
577    let lower: Vec<f64> = middle
578        .iter()
579        .zip(&atr_values)
580        .map(|(&m, &a)| {
581            if m.is_nan() || a.is_nan() {
582                f64::NAN
583            } else {
584                m - a * multiplier
585            }
586        })
587        .collect();
588
589    Ok((upper, middle, lower))
590}

592/// Donchian Channel
593///
594/// A price channel indicator showing the highest high and lowest low
595/// over a specified period.
596///
597/// # Algorithm
598/// ```text
599/// Upper Band  = MAX(high, period)
600/// Lower Band  = MIN(low, period)
601/// Middle Band = (Upper + Lower) / 2
602/// ```
603///
604/// # Parameters
605/// - `high`: High price series
606/// - `low`: Low price series
607/// - `period`: Lookback period (typically 20)
608///
609/// # Returns
610/// - `Ok((upper, middle, lower))`: Three vectors of channel values
611///
612/// # Errors
613/// - [`HazeError::EmptyInput`]: Any input array is empty
614/// - [`HazeError::LengthMismatch`]: Input arrays have different lengths
615/// - [`HazeError::InvalidPeriod`]: period is 0 or > data length
616///
617/// # Example
618/// ```rust
619/// use haze_library::indicators::volatility::donchian_channel;
620///
621/// let high = vec![102.0, 105.0, 104.0, 106.0, 103.0];
622/// let low = vec![99.0, 101.0, 100.0, 102.0, 98.0];
623///
624/// let (upper, middle, lower) = donchian_channel(&high, &low, 3).unwrap();
625/// assert_eq!(upper[2], 105.0);  // MAX of first 3 highs
626/// assert_eq!(lower[2], 99.0);   // MIN of first 3 lows
627/// assert_eq!(middle[2], 102.0); // (105 + 99) / 2
628/// ```
629pub fn donchian_channel(
630    high: &[f64],
631    low: &[f64],
632    period: usize,
633) -> HazeResult<(Vec<f64>, Vec<f64>, Vec<f64>)> {
634    // Validate inputs
635    validate_not_empty(high, "high")?;
636    validate_lengths_match(&[(high, "high"), (low, "low")])?;
637    validate_period(period, high.len())?;
638
639    let upper = rolling_max(high, period);
640    let lower = rolling_min(low, period);
641
642    let middle: Vec<f64> = upper
643        .iter()
644        .zip(&lower)
645        .map(|(&u, &l)| {
646            if u.is_nan() || l.is_nan() {
647                f64::NAN
648            } else {
649                (u + l) / 2.0
650            }
651        })
652        .collect();
653
654    Ok((upper, middle, lower))
655}

657/// Chandelier Exit
658///
659/// A volatility-based trailing stop system that uses ATR to set exit levels
660/// for both long and short positions.
661///
662/// # Algorithm
663/// ```text
664/// Long Exit  = MAX(high, period) - ATR(atr_period) * multiplier
665/// Short Exit = MIN(low, period) + ATR(atr_period) * multiplier
666/// ```
667///
668/// # Parameters
669/// - `high`: High price series
670/// - `low`: Low price series
671/// - `close`: Close price series
672/// - `period`: Lookback period for highest high / lowest low (typically 22)
673/// - `atr_period`: ATR period (typically 22)
674/// - `multiplier`: ATR multiplier for stop distance (typically 3.0)
675///
676/// # Returns
677/// - `Ok((long_exit, short_exit))`: Two vectors of exit levels
678///
679/// # Errors
680/// - [`HazeError::EmptyInput`]: Any input array is empty
681/// - [`HazeError::LengthMismatch`]: Input arrays have different lengths
682/// - [`HazeError::InvalidPeriod`]: period or atr_period is 0
683/// - [`HazeError::InsufficientData`]: period or atr_period > data length
684///
685/// # Example
686/// ```rust
687/// use haze_library::indicators::volatility::chandelier_exit;
688///
689/// let high = vec![102.0, 105.0, 104.0, 106.0, 108.0, 110.0];
690/// let low = vec![99.0, 101.0, 100.0, 102.0, 104.0, 106.0];
691/// let close = vec![101.0, 103.0, 102.0, 105.0, 107.0, 109.0];
692///
693/// let (long_exit, short_exit) = chandelier_exit(&high, &low, &close, 3, 3, 3.0).unwrap();
694/// ```
695pub fn chandelier_exit(
696    high: &[f64],
697    low: &[f64],
698    close: &[f64],
699    period: usize,
700    atr_period: usize,
701    multiplier: f64,
702) -> HazeResult<(Vec<f64>, Vec<f64>)> {
703    // Validate inputs
704    validate_not_empty(high, "high")?;
705    validate_lengths_match(&[(high, "high"), (low, "low"), (close, "close")])?;
706
707    let n = high.len();
708    validate_period(period, n)?;
709
710    // Validate ATR period
711    if atr_period == 0 {
712        return Err(HazeError::InvalidPeriod {
713            period: atr_period,
714            data_len: n,
715        });
716    }
717    if atr_period >= n {
718        return Err(HazeError::InsufficientData {
719            required: atr_period + 1,
720            actual: n,
721        });
722    }
723
724    // Validate multiplier
725    if multiplier < 0.0 {
726        return Err(HazeError::ParameterOutOfRange {
727            name: "multiplier",
728            value: multiplier,
729            min: 0.0,
730            max: f64::INFINITY,
731        });
732    }
733
734    let max_high = rolling_max(high, period);
735    let min_low = rolling_min(low, period);
736    let atr_values = atr(high, low, close, atr_period)?;
737
738    let mut long_exit = init_result!(n);
739    let mut short_exit = init_result!(n);
740
741    for i in 0..n {
742        let mh = max_high[i];
743        let ml = min_low[i];
744        let atr_val = atr_values[i];
745        if mh.is_nan() || ml.is_nan() || atr_val.is_nan() {
746            continue;
747        }
748        long_exit[i] = mh - atr_val * multiplier;
749        short_exit[i] = ml + atr_val * multiplier;
750    }
751
752    Ok((long_exit, short_exit))
753}

755/// Historical Volatility (HV)
756///
757/// Measures the annualized standard deviation of logarithmic returns,
758/// commonly used in options pricing and risk assessment.
759///
760/// # Algorithm
761/// ```text
762/// log_return[i] = ln(close[i] / close[i-1])
763/// HV = StdDev(log_returns, period) * sqrt(period) * 100
764/// ```
765///
766/// # Parameters
767/// - `close`: Close price series
768/// - `period`: Lookback period (typically 20)
769///
770/// # Returns
771/// - `Ok(Vec<f64>)`: Historical volatility as percentage
772///
773/// # Errors
774/// - [`HazeError::EmptyInput`]: close array is empty
775/// - [`HazeError::InvalidPeriod`]: period < 2 or > data length
776///
777/// # Example
778/// ```rust
779/// use haze_library::indicators::volatility::historical_volatility;
780///
781/// let close: Vec<f64> = (100..120).map(|x| x as f64).collect();
782/// let hv = historical_volatility(&close, 5).unwrap();
783/// ```
784pub fn historical_volatility(close: &[f64], period: usize) -> HazeResult<Vec<f64>> {
785    // Validate inputs
786    validate_not_empty(close, "close")?;
787
788    let n = close.len();
789
790    // Historical volatility requires period >= 2 for meaningful std dev
791    if period < 2 {
792        return Err(HazeError::InvalidPeriod {
793            period,
794            data_len: n,
795        });
796    }
797    validate_period(period, n)?;
798
799    let mut log_returns = init_result!(n);
800    for i in 1..n {
801        if is_zero(close[i - 1]) || close[i].is_nan() || close[i - 1].is_nan() {
802            continue;
803        }
804        log_returns[i] = (close[i] / close[i - 1]).ln();
805    }
806
807    let stdev_values = stdev(&log_returns, period);
808    let scale = (period as f64).sqrt() * 100.0;
809
810    let result = stdev_values
811        .iter()
812        .map(|&v| if v.is_nan() { f64::NAN } else { v * scale })
813        .collect();
814
815    Ok(result)
816}

818/// Ulcer Index
819///
820/// A volatility measure focusing on downside risk by measuring the depth
821/// and duration of drawdowns from recent highs.
822///
823/// # Algorithm
824/// ```text
825/// drawdown[i] = ((close[i] - rolling_max[i]) / rolling_max[i]) * 100
826/// Ulcer Index[i] = sqrt(mean(drawdown[i-period+1..i]^2))
827/// ```
828///
829/// # Parameters
830/// - `close`: Close price series
831/// - `period`: Lookback period (typically 14)
832///
833/// # Returns
834/// - `Ok(Vec<f64>)`: Ulcer Index values
835///
836/// # Errors
837/// - [`HazeError::EmptyInput`]: close array is empty
838/// - [`HazeError::InvalidPeriod`]: period is 0 or > data length
839///
840/// # Example
841/// ```rust
842/// use haze_library::indicators::volatility::ulcer_index;
843///
844/// let close = vec![100.0, 98.0, 99.0, 97.0, 100.0, 102.0];
845/// let ui = ulcer_index(&close, 3).unwrap();
846/// ```
847pub fn ulcer_index(close: &[f64], period: usize) -> HazeResult<Vec<f64>> {
848    // Validate inputs
849    validate_not_empty(close, "close")?;
850    validate_period(period, close.len())?;
851
852    let n = close.len();
853    let rolling_max_close = rolling_max(close, period);
854
855    let mut dd_sq = init_result!(n);
856    for i in 0..n {
857        let max_close = rolling_max_close[i];
858        let c = close[i];
859        if max_close.is_nan() || c.is_nan() || max_close <= 0.0 {
860            continue;
861        }
862        let dd = (c - max_close) / max_close * 100.0;
863        dd_sq[i] = dd * dd;
864    }
865
866    let dd_sq_clean: Vec<f64> = dd_sq
867        .iter()
868        .map(|&v| if v.is_finite() { v } else { 0.0 })
869        .collect();
870    let dd_sq_sum = rolling_sum_kahan(&dd_sq_clean, period);
871
872    let mut valid_prefix = vec![0usize; n + 1];
873    for i in 0..n {
874        let valid = if dd_sq[i].is_finite() { 1 } else { 0 };
875        valid_prefix[i + 1] = valid_prefix[i] + valid;
876    }
877
878    let mut result = init_result!(n);
879    for i in (period - 1)..n {
880        let valid_count = valid_prefix[i + 1] - valid_prefix[i + 1 - period];
881        if valid_count == period {
882            result[i] = (dd_sq_sum[i] / period as f64).sqrt();
883        }
884    }
885
886    Ok(result)
887}

889/// Mass Index
890///
891/// A volatility indicator that uses the ratio of two EMAs of the high-low range
892/// to identify potential trend reversals.
893///
894/// # Algorithm
895/// ```text
896/// Range = high - low
897/// EMA1  = EMA(Range, fast)
898/// EMA2  = EMA(EMA1, fast)
899/// Ratio = EMA1 / EMA2
900/// Mass Index = Sum(Ratio, slow)
901/// ```
902///
903/// A "reversal bulge" occurs when Mass Index rises above 27 and then falls below 26.5.
904///
905/// # Parameters
906/// - `high`: High price series
907/// - `low`: Low price series
908/// - `fast`: EMA period for smoothing (typically 9)
909/// - `slow`: Sum period (typically 25)
910///
911/// # Returns
912/// - `Ok(Vec<f64>)`: Mass Index values
913///
914/// # Errors
915/// - [`HazeError::EmptyInput`]: Any input array is empty
916/// - [`HazeError::LengthMismatch`]: Input arrays have different lengths
917/// - [`HazeError::InvalidPeriod`]: fast or slow is 0
918/// - [`HazeError::InsufficientData`]: data length is insufficient for fast/slow
919///
920/// # Example
921/// ```rust
922/// use haze_library::indicators::volatility::mass_index;
923///
924/// let high: Vec<f64> = (100..160).map(|x| x as f64 + 5.0).collect();
925/// let low: Vec<f64> = (100..160).map(|x| x as f64).collect();
926///
927/// let mi = mass_index(&high, &low, 9, 25).unwrap();
928/// ```
929pub fn mass_index(high: &[f64], low: &[f64], fast: usize, slow: usize) -> HazeResult<Vec<f64>> {
930    // Validate inputs
931    validate_not_empty(high, "high")?;
932    validate_lengths_match(&[(high, "high"), (low, "low")])?;
933
934    let n = high.len();
935    if fast == 0 {
936        return Err(HazeError::InvalidPeriod {
937            period: fast,
938            data_len: n,
939        });
940    }
941    if slow == 0 {
942        return Err(HazeError::InvalidPeriod {
943            period: slow,
944            data_len: n,
945        });
946    }
947    let (fast, slow) = if slow < fast {
948        (slow, fast)
949    } else {
950        (fast, slow)
951    };
952    let required = 2 * slow - fast;
953    validate_min_length(high, required)?;
954
955    let range: Vec<f64> = (0..n).map(|i| high[i] - low[i]).collect();
956    let ema1 = ema(&range, fast)?;
957    let ema2 = ema_allow_nan(&ema1, fast)?;
958
959    let ratio: Vec<f64> = ema1
960        .iter()
961        .zip(&ema2)
962        .map(|(&e1, &e2)| {
963            if e1.is_nan() || e2.is_nan() || is_zero(e2) {
964                f64::NAN
965            } else {
966                e1 / e2
967            }
968        })
969        .collect();
970
971    let ratio_clean: Vec<f64> = ratio
972        .iter()
973        .map(|&v| if v.is_finite() { v } else { 0.0 })
974        .collect();
975    let ratio_sum = rolling_sum_kahan(&ratio_clean, slow);
976
977    let mut valid_prefix = vec![0usize; n + 1];
978    for i in 0..n {
979        let valid = if ratio[i].is_finite() { 1 } else { 0 };
980        valid_prefix[i + 1] = valid_prefix[i] + valid;
981    }
982
983    let mut result = init_result!(n);
984    for i in (slow - 1)..n {
985        let valid_count = valid_prefix[i + 1] - valid_prefix[i + 1 - slow];
986        if valid_count == slow {
987            result[i] = ratio_sum[i];
988        }
989    }
990
991    Ok(result)
992}
