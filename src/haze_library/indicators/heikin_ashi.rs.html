1// indicators/heikin_ashi.rs - Heikin Ashi Candlesticks
2//
3// Heikin Ashi（平均K线）是一种平滑蜡烛图技术
4// - 用于过滤市场噪音
5// - 更清晰地显示趋势
6// - 减少假信号
7
8use crate::errors::validation::{validate_lengths_match, validate_not_empty};
9use crate::errors::{HazeError, HazeResult};

11/// Heikin Ashi 蜡烛图数据
12#[derive(Debug, Clone)]
13pub struct HeikinAshiCandles {
14    pub ha_open: Vec<f64>,
15    pub ha_high: Vec<f64>,
16    pub ha_low: Vec<f64>,
17    pub ha_close: Vec<f64>,
18}
19
20/// Heikin Ashi 信号结果
21#[derive(Debug, Clone)]
22pub struct HeikinAshiSignals {
23    pub candles: HeikinAshiCandles,
24    pub buy_signals: Vec<f64>,    // 买入信号（1.0 = 强烈买入）
25    pub sell_signals: Vec<f64>,   // 卖出信号（1.0 = 强烈卖出）
26    pub trend_strength: Vec<f64>, // 趋势强度（0.0-1.0）
27}
28
29/// 计算 Heikin Ashi 蜡烛图
30///
31/// # 参数
32/// - `open`: 开盘价
33/// - `high`: 最高价
34/// - `low`: 最低价
35/// - `close`: 收盘价
36///
37/// # 返回
38/// HeikinAshiCandles 包含平滑后的 OHLC 数据
39///
40/// # 算法
41/// ```text
42/// HA_Close = (O + H + L + C) / 4
43/// HA_Open = (HA_Open[prev] + HA_Close[prev]) / 2
44/// HA_High = max(H, HA_Open, HA_Close)
45/// HA_Low = min(L, HA_Open, HA_Close)
46/// ```
47pub fn heikin_ashi(
48    open: &[f64],
49    high: &[f64],
50    low: &[f64],
51    close: &[f64],
52) -> HazeResult<HeikinAshiCandles> {
53    validate_not_empty(close, "close")?;
54    validate_lengths_match(&[
55        (open, "open"),
56        (high, "high"),
57        (low, "low"),
58        (close, "close"),
59    ])?;
60
61    let len = close.len();
62    let mut ha_open = vec![0.0; len];
63    let mut ha_high = vec![0.0; len];
64    let mut ha_low = vec![0.0; len];
65    let mut ha_close = vec![0.0; len];
66
67    // 第一根蜡烛
68    ha_open[0] = (open[0] + close[0]) / 2.0;
69    ha_close[0] = (open[0] + high[0] + low[0] + close[0]) / 4.0;
70    ha_high[0] = high[0];
71    ha_low[0] = low[0];
72
73    // 后续蜡烛
74    for i in 1..len {
75        // HA Close
76        ha_close[i] = (open[i] + high[i] + low[i] + close[i]) / 4.0;
77
78        // HA Open
79        ha_open[i] = (ha_open[i - 1] + ha_close[i - 1]) / 2.0;
80
81        // HA High
82        ha_high[i] = high[i].max(ha_open[i]).max(ha_close[i]);
83
84        // HA Low
85        ha_low[i] = low[i].min(ha_open[i]).min(ha_close[i]);
86    }
87
88    Ok(HeikinAshiCandles {
89        ha_open,
90        ha_high,
91        ha_low,
92        ha_close,
93    })
94}

96/// Heikin Ashi 趋势信号
97///
98/// # 参数
99/// - `open`, `high`, `low`, `close`: 原始OHLC数据
100/// - `lookback`: 回看周期，用于判断连续趋势（默认 3）
101///
102/// # 返回
103/// HeikinAshiSignals 包含买卖信号和趋势强度
104///
105/// # 信号规则
106/// - 买入：连续N根看涨HA蜡烛（HA_Close > HA_Open，无下影线）
107/// - 卖出：连续N根看跌HA蜡烛（HA_Close < HA_Open，无上影线）
108/// - 趋势强度：蜡烛实体大小 / 蜡烛总高度
109pub fn heikin_ashi_signals(
110    open: &[f64],
111    high: &[f64],
112    low: &[f64],
113    close: &[f64],
114    lookback: usize,
115) -> HazeResult<HeikinAshiSignals> {
116    let candles = heikin_ashi(open, high, low, close)?;
117    let len = close.len();
118
119    let mut buy_signals = vec![0.0; len];
120    let mut sell_signals = vec![0.0; len];
121    let mut trend_strength = vec![0.0; len];
122
123    for i in lookback..len {
124        // 计算当前蜡烛的趋势强度
125        let ha_body = (candles.ha_close[i] - candles.ha_open[i]).abs();
126        let ha_range = candles.ha_high[i] - candles.ha_low[i];
127
128        if ha_range > 0.0 {
129            trend_strength[i] = ha_body / ha_range;
130        }
131
132        // 检查连续趋势
133        let mut bullish_count = 0;
134        let mut bearish_count = 0;
135
136        for j in 0..lookback {
137            let idx = i - j;
138            let is_bullish = candles.ha_close[idx] > candles.ha_open[idx];
139            let is_bearish = candles.ha_close[idx] < candles.ha_open[idx];
140
141            // 检查是否有下影线（看涨蜡烛）或上影线（看跌蜡烛）
142            let lower_shadow =
143                candles.ha_open[idx].min(candles.ha_close[idx]) - candles.ha_low[idx];
144            let upper_shadow =
145                candles.ha_high[idx] - candles.ha_open[idx].max(candles.ha_close[idx]);
146
147            if is_bullish && lower_shadow < ha_range * 0.1 {
148                bullish_count += 1;
149            }
150
151            if is_bearish && upper_shadow < ha_range * 0.1 {
152                bearish_count += 1;
153            }
154        }
155
156        // 生成信号
157        if bullish_count >= lookback {
158            // 连续看涨蜡烛
159            buy_signals[i] = 1.0;
160        } else if bullish_count >= lookback / 2 {
161            // 部分看涨
162            buy_signals[i] = 0.5;
163        }
164
165        if bearish_count >= lookback {
166            // 连续看跌蜡烛
167            sell_signals[i] = 1.0;
168        } else if bearish_count >= lookback / 2 {
169            // 部分看跌
170            sell_signals[i] = 0.5;
171        }
172    }
173
174    Ok(HeikinAshiSignals {
175        candles,
176        buy_signals,
177        sell_signals,
178        trend_strength,
179    })
180}

182#[cfg(test)]
183mod tests {
184    use super::*;
185
186    #[test]
187    fn test_heikin_ashi_basic() {
188        let open = vec![100.0, 101.0, 102.0, 103.0, 104.0];
189        let high = vec![102.0, 103.0, 104.0, 105.0, 106.0];
190        let low = vec![99.0, 100.0, 101.0, 102.0, 103.0];
191        let close = vec![101.0, 102.0, 103.0, 104.0, 105.0];
192
193        let result = heikin_ashi(&open, &high, &low, &close).unwrap();
194
195        // HA蜡烛应该平滑原始数据
196        assert_eq!(result.ha_open.len(), 5);
197        assert_eq!(result.ha_close.len(), 5);
198
199        // 第一根蜡烛
200        assert!((result.ha_open[0] - 100.5).abs() < 0.01);
201
202        // 后续蜡烛应该使用前一根的HA值
203        assert!(result.ha_open[1].is_finite());
204    }
205
206    #[test]
207    fn test_heikin_ashi_signals_uptrend() {
208        // 模拟上升趋势
209        let len = 20;
210        let open: Vec<f64> = (0..len).map(|i| 100.0 + i as f64).collect();
211        let high: Vec<f64> = (0..len).map(|i| 100.0 + i as f64 + 2.0).collect();
212        let low: Vec<f64> = (0..len).map(|i| 100.0 + i as f64 - 1.0).collect();
213        let close: Vec<f64> = (0..len).map(|i| 100.0 + i as f64 + 1.0).collect();
214
215        let result = heikin_ashi_signals(&open, &high, &low, &close, 3).unwrap();
216
217        // 上升趋势应该有买入信号
218        let buy_count: usize = result.buy_signals.iter().filter(|&&x| x > 0.0).count();
219
220        assert!(buy_count > 0, "Expected buy signals in uptrend");
221    }
222
223    #[test]
224    fn test_heikin_ashi_signals_downtrend() {
225        // 模拟下降趋势
226        let len = 20;
227        let open: Vec<f64> = (0..len).map(|i| 120.0 - i as f64).collect();

230        let close: Vec<f64> = (0..len).map(|i| 120.0 - i as f64 - 1.0).collect();
231
232        let result = heikin_ashi_signals(&open, &high, &low, &close, 3).unwrap();
233
234        // 下降趋势应该有卖出信号

236
237        assert!(sell_count > 0, "Expected sell signals in downtrend");
238    }
