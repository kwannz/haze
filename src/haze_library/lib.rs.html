211/// Macro for fail-fast error handling in Python FFI.
212///
213/// Converts `HazeResult<T>` to `PyResult<T>` and propagates all errors
214/// to Python without NaN fallbacks.
215///
216/// # Usage
217/// ```ignore
218/// // Single vector
219/// ok_or_nan!(result, len)
220///
221/// // Tuple of 2 vectors
222/// ok_or_nan!(result, len, 2)
223///
224/// // Tuple of 3 vectors
225/// ok_or_nan!(result, len, 3)
226/// ```
227#[cfg(feature = "python")]
228macro_rules! ok_or_nan {
229    // Single vector case: ok_or_nan!(result, len)
230    ($result:expr, $len:expr) => {{
231        let _ = $len;
232        match $result {
233            Ok(values) => Ok(values),
234            Err(err) => Err(err.into()),
235        }
236    }};
237    // 2-tuple case: ok_or_nan!(result, len, 2)
238    ($result:expr, $len:expr, 2) => {{
239        let _ = $len;
240        match $result {
241            Ok(values) => Ok(values),
242            Err(err) => Err(err.into()),
243        }
244    }};
245    // 3-tuple case: ok_or_nan!(result, len, 3)
246    ($result:expr, $len:expr, 3) => {{
247        let _ = $len;
248        match $result {
249            Ok(values) => Ok(values),
250            Err(err) => Err(err.into()),
251        }
252    }};
253    // 4-tuple case: ok_or_nan!(result, len, 4)
254    ($result:expr, $len:expr, 4) => {{
255        let _ = $len;
256        match $result {
257            Ok(values) => Ok(values),
258            Err(err) => Err(err.into()),
259        }
260    }};
261    // 5-tuple case: ok_or_nan!(result, len, 5)
262    ($result:expr, $len:expr, 5) => {{
263        let _ = $len;
264        match $result {
265            Ok(values) => Ok(values),
266            Err(err) => Err(err.into()),
267        }
268    }};
269    // 6-tuple case: ok_or_nan!(result, len, 6)
270    ($result:expr, $len:expr, 6) => {{
271        let _ = $len;
272        match $result {
273            Ok(values) => Ok(values),
274            Err(err) => Err(err.into()),
275        }
276    }};
277    // 7-tuple case: ok_or_nan!(result, len, 7)
278    ($result:expr, $len:expr, 7) => {{
279        let _ = $len;
280        match $result {
281            Ok(values) => Ok(values),
282            Err(err) => Err(err.into()),
283        }
284    }};
285    // Special case: 2 vectors + 1 f64: ok_or_nan!(result, len, 2, f64)
286    ($result:expr, $len:expr, 2, f64) => {{
287        let _ = $len;
288        match $result {
289            Ok(values) => Ok(values),
290            Err(err) => Err(err.into()),
291        }
292    }};
293}

6872/// Python-accessible ML model wrapper
6873#[cfg(feature = "python")]
6874#[pyclass(name = "SFGModel")]
6875pub struct PySFGModel {
6876    inner: ml::SFGModel,
6877    features_dim: usize,
6878}
6879
6880#[cfg(feature = "python")]
6881#[pymethods]
6882impl PySFGModel {
6883    /// Check if model is trained
6884    pub fn is_trained(&self) -> bool {
6885        self.inner.is_trained()
6886    }
6887
6888    /// Get feature dimension
6889    pub fn features_dim(&self) -> usize {
6890        self.features_dim
6891    }
6892
6893    /// Predict using the trained model
6894    ///
6895    /// # Arguments
6896    /// * `features` - Flattened feature array (n_samples * n_features)
6897    /// * `n_samples` - Number of samples
6898    ///
6899    /// # Returns
6900    /// Predictions for each sample
6901    pub fn predict(&self, features: Vec<f64>, n_samples: usize) -> PyResult<Vec<f64>> {
6902        if !self.inner.is_trained() {
6903            return Err(pyo3::exceptions::PyValueError::new_err(
6904                "Model is not trained",
6905            ));
6906        }
6907
6908        let n_features = self.features_dim;
6909        if features.len() != n_samples * n_features {
6910            return Err(pyo3::exceptions::PyValueError::new_err(format!(
6911                "Expected {} features ({}x{}), got {}",
6912                n_samples * n_features,
6913                n_samples,
6914                n_features,
6915                features.len()
6916            )));
6917        }
6918
6919        let features_array = ndarray::Array2::from_shape_vec((n_samples, n_features), features)
6920            .map_err(|e| pyo3::exceptions::PyValueError::new_err(format!("{e}")))?;
6921
6922        Ok(self.inner.predict(&features_array).to_vec())
6923    }
6924}
