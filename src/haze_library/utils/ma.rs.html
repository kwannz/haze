1//! Moving Average Utilities Module
2//!
3//! # Overview
4//! This module provides fundamental moving average calculations that serve as
5//! building blocks for other technical indicators. All MA functions are optimized
6//! for performance with incremental updates and warmup NaNs.
7//!
8//! # Available Functions
9//! - [`sma`] - Simple Moving Average (arithmetic mean)
10//! - [`ema`] - Exponential Moving Average (weighted recent values)
11//! - [`rma`] - Wilder's Moving Average / Running Moving Average
12//! - [`wma`] - Weighted Moving Average (linear weights)
13//! - [`dema`] - Double Exponential Moving Average (reduced lag)
14//! - [`tema`] - Triple Exponential Moving Average (further lag reduction)
15//! - [`kama`] - Kaufman Adaptive Moving Average (volatility-adjusted)
16//! - `zlema` - Zero-Lag Exponential Moving Average (lag compensation)
17//! - [`hma`] - Hull Moving Average (smoothness with reduced lag)
18//! - `vwma` - Volume Weighted Moving Average
19//! - [`vwap`] - Volume Weighted Average Price
20//!
21//! # Usage Patterns
22//! These functions are typically used:
23//! - Directly for trend identification (crossover signals)
24//! - As components in composite indicators (MACD uses EMA, ATR uses RMA)
25//! - For smoothing noisy price data
26//!
27//! # Examples
28//! ```rust
29//! use haze_library::utils::ma::{sma, ema, rma};
30//!
31//! let prices = vec![100.0, 101.0, 102.0, 103.0, 104.0, 105.0];
32//!
33//! // Simple Moving Average (3-period)
34//! let sma_values = sma(&prices, 3);
35//!
36//! // Exponential Moving Average (more weight on recent prices)
37//! let ema_values = ema(&prices, 3);
38//!
39//! // Wilder's RMA (used in ATR, RSI calculations)
40//! let rma_values = rma(&prices, 3);
41//! ```
42//!
43//! # Performance Characteristics
44//! - SMA: O(n) with incremental sum updates
45//! - EMA/RMA: O(n) single pass with exponential decay
46//! - WMA: O(n * period) due to weighted summation
47//! - DEMA/TEMA: O(n) with chained EMA calculations
48//!
49//! # NaN Handling
50//! - Inputs must be finite; NaN/Inf are rejected by validation
51//! - Warmup periods (first period-1 values) are NaN by design
52//! - Output maintains same length as input array
53//!
54//! # Cross-References
55//! - [`crate::indicators::momentum`] - MACD uses EMA
56//! - [`crate::indicators::volatility`] - ATR uses RMA
57//! - [`crate::indicators::trend`] - SuperTrend uses ATR (via RMA)

68/// SMA - Simple Moving Average（简单移动平均）
69///
70/// 算法：sum(values[i-period+1 .. i+1]) / period
71///
72/// 使用 Kahan 补偿求和的增量更新并定期重新计算以防止浮点误差累积。
73///
74/// # 参数
75/// - `values`: 输入序列
76/// - `period`: 周期
77///
78/// # 返回
79/// - 与输入等长的向量，前 period-1 个值为 NaN
80///
81/// # 错误
82/// - 如果输入为空，返回 `HazeError::EmptyInput`
83/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
84fn sma_impl(values: &[f64], period: usize) -> Vec<f64> {
85    /// 重新计算间隔：每 1000 次迭代重新计算一次以重置累积误差
86    const RECALC_INTERVAL: usize = 1000;
87
88    let n = values.len();
89    let mut result = init_result!(n);
90    let mut sum = 0.0;
91    let mut compensation = 0.0; // Kahan 补偿项
92    let mut count = 0usize;
93    let mut steps_since_recalc = 0usize;
94
95    for i in 0..n {
96        let v = values[i];
97        if v.is_nan() {
98            sum = 0.0;
99            compensation = 0.0;
100            count = 0;
101            steps_since_recalc = 0;
102            continue;
103        }
104
105        // 使用 Kahan 求和添加新值
106        let y = v - compensation;
107        let t = sum + y;
108        compensation = (t - sum) - y;
109        sum = t;
110        count += 1;
111
112        if count > period {
113            // 使用 Kahan 求和减去旧值
114            let old_val = values[i - period];
115            let y = -old_val - compensation;
116            let t = sum + y;
117            compensation = (t - sum) - y;
118            sum = t;
119            count = period;
120            steps_since_recalc += 1;
121
122            // 定期完整重新计算以消除累积浮点误差
123            if steps_since_recalc >= RECALC_INTERVAL {
124                sum = kahan_sum(&values[i + 1 - period..=i]);
125                compensation = 0.0;
126                steps_since_recalc = 0;
127            }
128        }
129
130        if count == period {
131            result[i] = sum / period as f64;
132        }
133    }
134
135    result
136}

151/// EMA - Exponential Moving Average（指数移动平均）
152///
153/// 算法：
154/// - alpha = 2 / (period + 1)
155/// - EMA`[0]` = SMA(period)  // 初始值使用 SMA
156/// - EMA`[i]` = alpha * value`[i]` + (1 - alpha) * EMA`[i-1]`
157///
158/// # 参数
159/// - `values`: 输入序列
160/// - `period`: 周期
161///
162/// # 返回
163/// - 与输入等长的向量，前 period-1 个值为 NaN
164///
165/// # 错误
166/// - 如果输入为空，返回 `HazeError::EmptyInput`
167/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
168fn ema_impl(values: &[f64], period: usize, alpha: f64) -> Vec<f64> {
169    let n = values.len();
170    let mut result = init_result!(n);
171    let mut warmup_sum = 0.0;
172    let mut warmup_comp = 0.0;
173    let mut warmup_count = 0usize;
174    let mut prev = f64::NAN;
175
176    for i in 0..n {
177        let v = values[i];
178        if v.is_nan() {
179            warmup_sum = 0.0;
180            warmup_comp = 0.0;
181            warmup_count = 0;
182            prev = f64::NAN;
183            continue;
184        }
185
186        if warmup_count < period {
187            warmup_count += 1;
188            let y = v - warmup_comp;
189            let t = warmup_sum + y;
190            warmup_comp = (t - warmup_sum) - y;
191            warmup_sum = t;
192
193            if warmup_count == period {
194                prev = warmup_sum / period as f64;
195                result[i] = prev;
196            }
197            continue;
198        }
199
200        prev = alpha * v + (1.0 - alpha) * prev;
201        result[i] = prev;
202    }
203
204    result
205}

256/// WMA - Weighted Moving Average（加权移动平均）
257///
258/// 使用 O(n) 增量算法实现。
259///
260/// 算法：WMA = sum(value`[i]` * weight`[i]`) / sum(weight)
261/// 其中 weight`[i]` = i + 1（线性递增权重）
262///
263/// # 增量更新原理
264/// 当窗口从 [v0, v1, ..., v_{n-1}] 滑动到 [v1, v2, ..., v_n] 时：
265/// - 所有现有值的权重都减 1（等于减去 simple_sum）
266/// - 旧值 v0（权重 1）被移除
267/// - 新值 v_n（权重 period）被添加
268///
269/// 公式：new_weighted_sum = old_weighted_sum - simple_sum + period * new_value
270///
271/// # 参数
272/// - `values`: 输入序列
273/// - `period`: 周期
274///
275/// # 返回
276/// - 与输入等长的向量，前 period-1 个值为 NaN
277///
278/// # 性能
279/// - 时间复杂度: O(n)
280/// - 空间复杂度: O(n)
281///
282/// # 错误
283/// - 如果输入为空，返回 `HazeError::EmptyInput`
284/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
285fn wma_impl(values: &[f64], period: usize) -> Vec<f64> {
286    const RECALC_INTERVAL: usize = 1000;
287
288    let n = values.len();
289
290    let period_f = period as f64;
291    let weight_sum = period_f * (period_f + 1.0) / 2.0;
292    let mut result = vec![f64::NAN; n];
293
294    // Kahan 补偿求和：提升增量更新的数值稳定性
295    #[inline]
296    fn kahan_add(sum: &mut f64, compensation: &mut f64, value: f64) {
297        let y = value - *compensation;
298        let t = *sum + y;
299        *compensation = (t - *sum) - y;
300        *sum = t;
301    }
302
303    let mut simple_sum = 0.0;
304    let mut simple_comp = 0.0;
305    let mut weighted_sum = 0.0;
306    let mut weighted_comp = 0.0;
307    let mut count = 0usize;
308    let mut steps_since_recalc = 0usize;
309
310    // 单次遍历：累积到 period 后使用 O(n) 增量更新
311    for i in 0..n {
312        let v = values[i];
313
314        if v.is_nan() {
315            simple_sum = 0.0;
316            simple_comp = 0.0;
317            weighted_sum = 0.0;
318            weighted_comp = 0.0;
319            count = 0;
320            steps_since_recalc = 0;
321            continue;
322        }
323
324        if count < period {
325            count += 1;
326            kahan_add(&mut simple_sum, &mut simple_comp, v);
327            kahan_add(&mut weighted_sum, &mut weighted_comp, (count as f64) * v);
328
329            if count == period {
330                result[i] = weighted_sum / weight_sum;
331            }
332            continue;
333        }
334
335        // count == period: 增量更新当前窗口
336        if steps_since_recalc >= RECALC_INTERVAL {
337            // 定期重新计算以防止浮点误差累积（同时保证遇到历史 NaN 后可恢复）
338            simple_sum = 0.0;
339            simple_comp = 0.0;
340            weighted_sum = 0.0;
341            weighted_comp = 0.0;
342
343            let window = &values[i + 1 - period..=i];
344            for (j, &val) in window.iter().enumerate() {
345                kahan_add(&mut simple_sum, &mut simple_comp, val);
346                kahan_add(
347                    &mut weighted_sum,
348                    &mut weighted_comp,
349                    (j as f64 + 1.0) * val,
350                );
351            }
352            steps_since_recalc = 0;
353        } else {
354            let prev_simple_sum = simple_sum;
355
356            // new_weighted_sum = old_weighted_sum - simple_sum + period * new_value
357            kahan_add(&mut weighted_sum, &mut weighted_comp, -prev_simple_sum);
358            kahan_add(&mut weighted_sum, &mut weighted_comp, period_f * v);
359
360            // new_simple_sum = old_simple_sum - old_value + new_value
361            let old = values[i - period];
362            kahan_add(&mut simple_sum, &mut simple_comp, -old);
363            kahan_add(&mut simple_sum, &mut simple_comp, v);
364        }
365
366        result[i] = weighted_sum / weight_sum;
367        steps_since_recalc += 1;
368    }
369
370    result
371}

386/// HMA - Hull Moving Average（赫尔移动平均，低延迟）
387///
388/// 算法：
389/// - half_period = period / 2
390/// - sqrt_period = sqrt(period)
391/// - HMA = WMA(2 * WMA(half_period) - WMA(period), sqrt_period)
392///
393/// 特点：响应速度快，延迟低
394///
395/// # 参数
396/// - `values`: 输入序列
397/// - `period`: 周期
398///
399/// # 返回
400/// - 与输入等长的向量
401///
402/// # 错误
403/// - 如果输入为空，返回 `HazeError::EmptyInput`
404/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
405pub fn hma(values: &[f64], period: usize) -> HazeResult<Vec<f64>> {
406    // Fail-Fast 验证
407    validate_not_empty(values, "values")?;
408    validate_period(period, values.len())?;
409
410    let half_period = period / 2;
411    let sqrt_period = (period as f64).sqrt() as usize;
412
413    let wma_half = wma_allow_nan(values, half_period)?;
414    let wma_full = wma_allow_nan(values, period)?;
415
416    // 2 * WMA(half) - WMA(full)
417    let diff: Vec<f64> = wma_half
418        .iter()
419        .zip(&wma_full)
420        .map(|(&a, &b)| {
421            if a.is_nan() || b.is_nan() {
422                f64::NAN
423            } else {
424                2.0 * a - b
425            }
426        })
427        .collect();
428
429    let result = wma_allow_nan(&diff, sqrt_period)?;
430    if result.iter().all(|v| v.is_nan()) {
431        Ok(diff)
432    } else {
433        Ok(result)
434    }
435}

437/// DEMA - Double Exponential Moving Average（双重指数移动平均）
438///
439/// 算法：DEMA = 2 * EMA(period) - EMA(EMA(period))
440///
441/// # 参数
442/// - `values`: 输入序列
443/// - `period`: 周期
444///
445/// # 返回
446/// - 与输入等长的向量
447///
448/// # 错误
449/// - 如果输入为空，返回 `HazeError::EmptyInput`
450/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
451pub fn dema(values: &[f64], period: usize) -> HazeResult<Vec<f64>> {
452    // Fail-Fast 验证
453    validate_not_empty(values, "values")?;
454    validate_period(period, values.len())?;
455
456    let ema1 = ema_allow_nan(values, period)?;
457    let ema2 = ema_allow_nan(&ema1, period)?;
458
459    Ok(ema1
460        .iter()
461        .zip(&ema2)
462        .map(|(&e1, &e2)| {
463            if e1.is_nan() || e2.is_nan() {
464                f64::NAN
465            } else {
466                2.0 * e1 - e2
467            }
468        })
469        .collect())
470}

472/// TEMA - Triple Exponential Moving Average（三重指数移动平均）
473///
474/// 算法：TEMA = 3*EMA - 3*EMA(EMA) + EMA(EMA(EMA))
475///
476/// # 参数
477/// - `values`: 输入序列
478/// - `period`: 周期
479///
480/// # 返回
481/// - 与输入等长的向量
482///
483/// # 错误
484/// - 如果输入为空，返回 `HazeError::EmptyInput`
485/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
486pub fn tema(values: &[f64], period: usize) -> HazeResult<Vec<f64>> {
487    // Fail-Fast 验证
488    validate_not_empty(values, "values")?;
489    validate_period(period, values.len())?;
490
491    let ema1 = ema_allow_nan(values, period)?;
492    let ema2 = ema_allow_nan(&ema1, period)?;
493    let ema3 = ema_allow_nan(&ema2, period)?;
494
495    if !ema3.iter().any(|v| !v.is_nan()) {
496        return dema(values, period);
497    }
498
499    Ok(ema1
500        .iter()
501        .zip(&ema2)
502        .zip(&ema3)
503        .map(|((&e1, &e2), &e3)| {
504            if e1.is_nan() || e2.is_nan() || e3.is_nan() {
505                f64::NAN
506            } else {
507                3.0 * e1 - 3.0 * e2 + e3
508            }
509        })
510        .collect())
511}

513/// VWAP - Volume Weighted Average Price（成交量加权平均价）
514///
515/// 算法：VWAP = sum(typical_price * volume) / sum(volume)
516///
517/// 使用 Kahan 补偿求和的增量更新并定期重新计算以防止浮点误差累积。
518///
519/// # 参数
520/// - `typical_prices`: 典型价格序列 (H+L+C)/3
521/// - `volumes`: 成交量序列
522/// - `period`: 周期（0 表示累积 VWAP）
523///
524/// # 返回
525/// - 与输入等长的向量
526///
527/// # 错误
528/// - 如果输入为空，返回 `HazeError::EmptyInput`
529/// - 如果两个输入数组长度不匹配，返回 `HazeError::LengthMismatch`
530/// - 如果 period 超过数据长度，返回 `HazeError::InvalidPeriod`
531pub fn vwap(typical_prices: &[f64], volumes: &[f64], period: usize) -> HazeResult<Vec<f64>> {
532    /// 重新计算间隔：每 1000 次迭代重新计算一次以重置累积误差
533    const RECALC_INTERVAL: usize = 1000;
534
535    // Fail-Fast 验证
536    validate_not_empty(typical_prices, "typical_prices")?;
537    validate_same_length(typical_prices, "typical_prices", volumes, "volumes")?;
538
539    let n = typical_prices.len();
540    let mut result = init_result!(n);
541
542    if period == 0 {
543        // 累积 VWAP（使用 Kahan 求和算法减少浮点误差）
544        let mut cum_pv = 0.0;
545        let mut cum_v = 0.0;
546        let mut pv_comp = 0.0; // Kahan 补偿项
547        let mut v_comp = 0.0; // Kahan 补偿项
548        for i in 0..n {
549            // Kahan 求和 for pv
550            let pv = typical_prices[i] * volumes[i];
551            let pv_y = pv - pv_comp;
552            let pv_t = cum_pv + pv_y;
553            pv_comp = (pv_t - cum_pv) - pv_y;
554            cum_pv = pv_t;
555
556            // Kahan 求和 for v
557            let v = volumes[i];
558            let v_y = v - v_comp;
559            let v_t = cum_v + v_y;
560            v_comp = (v_t - cum_v) - v_y;
561            cum_v = v_t;
562
563            result[i] = if is_zero(cum_v) {
564                f64::NAN
565            } else {
566                cum_pv / cum_v
567            };
568        }
569        return Ok(result);
570    }
571
572    // 验证滚动 VWAP 的周期
573    if period > n {
574        validate_period(period, n)?;
575    }
576
577    // 滚动 VWAP（使用 Kahan 补偿的增量更新）
578    // 初始窗口使用 Kahan 求和
579    let mut pv_sum = 0.0;
580    let mut v_sum = 0.0;
581    let mut pv_comp = 0.0;
582    let mut v_comp = 0.0;
583
584    for i in 0..period {
585        let pv = typical_prices[i] * volumes[i];
586        let y = pv - pv_comp;
587        let t = pv_sum + y;
588        pv_comp = (t - pv_sum) - y;
589        pv_sum = t;
590
591        let y = volumes[i] - v_comp;
592        let t = v_sum + y;
593        v_comp = (t - v_sum) - y;
594        v_sum = t;
595    }
596
597    result[period - 1] = if is_zero(v_sum) {
598        f64::NAN
599    } else {
600        pv_sum / v_sum
601    };
602
603    for i in period..n {
604        // 定期完整重新计算以消除累积浮点误差
605        if (i - period + 1).is_multiple_of(RECALC_INTERVAL) {
606            pv_sum = 0.0;
607            v_sum = 0.0;
608            pv_comp = 0.0;
609            v_comp = 0.0;
610
611            for j in (i + 1 - period)..=i {
612                let pv = typical_prices[j] * volumes[j];
613                let y = pv - pv_comp;
614                let t = pv_sum + y;
615                pv_comp = (t - pv_sum) - y;
616                pv_sum = t;
617
618                let y = volumes[j] - v_comp;
619                let t = v_sum + y;
620                v_comp = (t - v_sum) - y;
621                v_sum = t;
622            }
623        } else {
624            // 使用 Kahan 补偿进行增量更新
625            // 添加新的 pv
626            let new_pv = typical_prices[i] * volumes[i];
627            let y = new_pv - pv_comp;
628            let t = pv_sum + y;
629            pv_comp = (t - pv_sum) - y;
630            pv_sum = t;
631
632            // 减去旧的 pv
633            let old_pv = typical_prices[i - period] * volumes[i - period];
634            let y = -old_pv - pv_comp;
635            let t = pv_sum + y;
636            pv_comp = (t - pv_sum) - y;
637            pv_sum = t;
638
639            // 添加新的 volume
640            let y = volumes[i] - v_comp;
641            let t = v_sum + y;
642            v_comp = (t - v_sum) - y;
643            v_sum = t;
644
645            // 减去旧的 volume
646            let y = -volumes[i - period] - v_comp;
647            let t = v_sum + y;
648            v_comp = (t - v_sum) - y;
649            v_sum = t;
650        }
651
652        result[i] = if is_zero(v_sum) {
653            f64::NAN
654        } else {
655            pv_sum / v_sum
656        };
657    }
658
659    Ok(result)
660}

744/// ZLMA (Zero Lag Moving Average) 零延迟移动平均
745///
746/// 尝试消除 EMA 的延迟，更快响应价格变化
747///
748/// - `values`: 输入序列
749/// - `period`: 周期
750///
751/// 返回：ZLMA 序列
752///
753/// # 算法
754/// 1. Lag = (period - 1) / 2
755/// 2. EMA_Data = 2 * values - values`[lag_ago]`
756/// 3. ZLMA = EMA(EMA_Data, period)
757///
758/// # 错误
759/// - 如果输入为空，返回 `HazeError::EmptyInput`
760/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
761pub fn zlma(values: &[f64], period: usize) -> HazeResult<Vec<f64>> {
762    // Fail-Fast 验证
763    validate_not_empty(values, "values")?;
764    validate_period(period, values.len())?;
765
766    let n = values.len();
767    let lag = (period - 1) / 2;
768    let mut ema_data = init_result!(n);
769
770    for i in lag..n {
771        ema_data[i] = 2.0 * values[i] - values[i - lag];
772    }
773
774    ema_allow_nan(&ema_data, period)
775}

777/// T3 (Tillson T3) 移动平均
778///
779/// 6 重 EMA 平滑，减少噪音同时保持快速响应
780///
781/// - `values`: 输入序列
782/// - `period`: 周期
783/// - `v_factor`: 平滑因子（通常 0.7）
784///
785/// 返回：T3 序列
786///
787/// # 算法
788/// 使用 6 层 EMA 和特殊系数
789///
790/// # 错误
791/// - 如果输入为空，返回 `HazeError::EmptyInput`
792/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
793pub fn t3(values: &[f64], period: usize, v_factor: f64) -> HazeResult<Vec<f64>> {
794    // Fail-Fast 验证
795    validate_not_empty(values, "values")?;
796    validate_period(period, values.len())?;
797
798    let n = values.len();
799
800    // 计算系数
801    let c1 = -v_factor * v_factor * v_factor;
802    let c2 = 3.0 * v_factor * v_factor + 3.0 * v_factor * v_factor * v_factor;
803    let c3 = -6.0 * v_factor * v_factor - 3.0 * v_factor - 3.0 * v_factor * v_factor * v_factor;
804    let c4 = 1.0 + 3.0 * v_factor + v_factor * v_factor * v_factor + 3.0 * v_factor * v_factor;
805
806    // 6 层 EMA
807    let e1 = ema_allow_nan(values, period)?;
808    let e2 = ema_allow_nan(&e1, period)?;
809    let e3 = ema_allow_nan(&e2, period)?;
810    let e4 = ema_allow_nan(&e3, period)?;
811    let e5 = ema_allow_nan(&e4, period)?;
812    let e6 = ema_allow_nan(&e5, period)?;
813
814    // 加权组合
815    let mut t3_values = init_result!(n);
816    for i in 0..n {
817        if !e3[i].is_nan() && !e4[i].is_nan() && !e5[i].is_nan() && !e6[i].is_nan() {
818            t3_values[i] = c1 * e6[i] + c2 * e5[i] + c3 * e4[i] + c4 * e3[i];
819        }
820    }
821
822    if t3_values.iter().all(|v| v.is_nan()) {
823        ema_allow_nan(values, period)
824    } else {
825        Ok(t3_values)
826    }
827}

829/// KAMA (Kaufman's Adaptive Moving Average) 考夫曼自适应移动平均
830///
831/// 根据市场波动性自适应调整平滑度
832///
833/// - `values`: 输入序列
834/// - `period`: 效率比率周期（默认 10）
835/// - `fast_period`: 快速 EMA 周期（默认 2）
836/// - `slow_period`: 慢速 EMA 周期（默认 30）
837///
838/// 返回：KAMA 序列
839///
840/// # 算法
841/// 1. Change = |Price`[i]` - Price[i-period]|
842/// 2. Volatility = Sum(|Price`[i]` - Price`[i-1]`|, period)
843/// 3. ER (Efficiency Ratio) = Change / Volatility
844/// 4. SC (Smoothing Constant) = [ER * (Fast_SC - Slow_SC) + Slow_SC]^2
845/// 5. KAMA`[i]` = KAMA`[i-1]` + SC * (Price`[i]` - KAMA`[i-1]`)
846///
847/// # 错误
848/// - 如果输入为空，返回 `HazeError::EmptyInput`
849/// - 如果 period 为 0 或超过数据长度，返回 `HazeError::InvalidPeriod`
850pub fn kama(
851    values: &[f64],
852    period: usize,
853    fast_period: usize,
854    slow_period: usize,
855) -> HazeResult<Vec<f64>> {
856    // Fail-Fast 验证
857    validate_not_empty(values, "values")?;
858    validate_period(period, values.len())?;
859
860    let n = values.len();
861
862    // 计算 EMA 平滑常数
863    let fast_sc = 2.0 / (fast_period + 1) as f64;
864    let slow_sc = 2.0 / (slow_period + 1) as f64;
865
866    let mut kama_values = init_result!(n);
867    kama_values[period - 1] = values[period - 1]; // 初始值
868
869    for i in period..n {
870        // 1. 计算价格变化
871        let change = (values[i] - values[i - period]).abs();
872
873        // 2. 计算波动性（价格变动的绝对值和）
874        let mut volatility = 0.0;
875        for j in 0..period {
876            let idx = i - period + 1 + j;
877            volatility += (values[idx] - values[idx - 1]).abs();
878        }
879
880        // 3. 效率比率
881        let er = if volatility > 0.0 {
882            change / volatility
883        } else {
884            0.0
885        };
886
887        // 4. 平滑常数
888        let sc = (er * (fast_sc - slow_sc) + slow_sc).powi(2);
889
890        // 5. KAMA
891        kama_values[i] = kama_values[i - 1] + sc * (values[i] - kama_values[i - 1]);
892    }
893
894    Ok(kama_values)
895}

897/// FRAMA (Fractal Adaptive Moving Average) 分形自适应移动平均
898///
899/// 基于分形维度自适应调整
900///
901/// - `values`: 输入序列
902/// - `period`: 周期（必须是偶数，默认 16）
903///
904/// 返回：FRAMA 序列
905///
906/// # 算法
907/// 使用分形维度计算自适应 alpha
908///
909/// # 错误
910/// - 如果输入为空，返回 `HazeError::EmptyInput`
911/// - 如果 period < 2 或非偶数或超过数据长度，返回 `HazeError::InvalidPeriod`
912pub fn frama(values: &[f64], period: usize) -> HazeResult<Vec<f64>> {
913    // Fail-Fast 验证
914    validate_not_empty(values, "values")?;
915
916    let n = values.len();
917    if period < 2 || !period.is_multiple_of(2) || period > n {
918        return Err(HazeError::InvalidPeriod {
919            period,
920            data_len: n,
921        });
922    }
923
924    let half = period / 2;
925    let mut frama_values = init_result!(n);
926
927    // 初始值
928    frama_values[period - 1] = values[period - 1];
929
930    for i in period..n {
931        // 计算前半周期和后半周期的最高最低价
932        let mut n1_high = f64::NEG_INFINITY;
933        let mut n1_low = f64::INFINITY;
934        let mut n2_high = f64::NEG_INFINITY;
935        let mut n2_low = f64::INFINITY;
936
937        for j in 0..half {
938            let idx1 = i - period + j;
939            let idx2 = i - half + j;
940
941            n1_high = n1_high.max(values[idx1]);
942            n1_low = n1_low.min(values[idx1]);
943            n2_high = n2_high.max(values[idx2]);
944            n2_low = n2_low.min(values[idx2]);
945        }
946
947        let n1 = (n1_high - n1_low) / (half as f64);
948        let n2 = (n2_high - n2_low) / (half as f64);
949
950        let mut n3_high = f64::NEG_INFINITY;
951        let mut n3_low = f64::INFINITY;
952        for j in 0..period {
953            let idx = i - period + j;
954            n3_high = n3_high.max(values[idx]);
955            n3_low = n3_low.min(values[idx]);
956        }
957        let n3 = (n3_high - n3_low) / (period as f64);
958
959        // 分形维度
960        let dimen = if n1 + n2 > 0.0 && n3 > 0.0 {
961            ((n1 + n2).ln() - n3.ln()) / 2_f64.ln()
962        } else {
963            1.0
964        };
965
966        // Alpha
967        let alpha = (-4.6 * (dimen - 1.0)).exp();
968        let alpha_clamped = alpha.clamp(0.01, 1.0);
969
970        // FRAMA
971        frama_values[i] = alpha_clamped * values[i] + (1.0 - alpha_clamped) * frama_values[i - 1];
972    }
973
974    Ok(frama_values)
975}
